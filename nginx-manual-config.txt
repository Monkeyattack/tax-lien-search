# Manual Nginx Configuration for tax.profithits.app
# Add this server block to your existing nginx configuration

# SSH into your VPS server: ssh root@172.93.51.42

# 1. First, create the site configuration file:
sudo nano /etc/nginx/sites-available/tax-profithits

# 2. Add this configuration:

server {
    listen 80;
    server_name tax.profithits.app;
    
    # Root directory for React build files
    root /var/www/tax-lien-search/frontend/build;
    index index.html index.htm;
    
    # Enable gzip compression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml text/javascript application/javascript application/xml+rss application/json;
    
    # Handle client-side routing (React Router)
    location / {
        try_files $uri $uri/ /index.html;
    }
    
    # API proxy to backend (running on port 8000)
    location /api {
        proxy_pass http://localhost:8000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
    
    # Static assets caching
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
    
    # Security - deny access to hidden files
    location ~ /\. {
        deny all;
    }
    
    # Logging
    access_log /var/log/nginx/tax-profithits-access.log;
    error_log /var/log/nginx/tax-profithits-error.log;
}

# 3. Enable the site:
sudo ln -s /etc/nginx/sites-available/tax-profithits /etc/nginx/sites-enabled/

# 4. Test nginx configuration:
sudo nginx -t

# 5. Reload nginx:
sudo systemctl reload nginx

# 6. Create the web directory:
sudo mkdir -p /var/www/tax-lien-search/frontend/build
sudo chown -R www-data:www-data /var/www/tax-lien-search

# scp -r build/* root@172.93.51.42:/var/www/tax-lien-search/frontend/build/
